1   package pacman.pacman.leveleditor;
2   
3   import levelmanagement.LevelRunner;
4   import java.awt.CardLayout;
5   import java.awt.Dimension;
6   import javax.swing.JFrame;
7   import javax.swing.JPanel;
8   
9   /*
10   * WindowHandler takes care of showing the right content on the screen
11   */
12  public class WindowHandler implements Runnable {
13  
14      private CardLayout cardlayout;
15      private JPanel cardPanel;
16  
17      public WindowHandler() {
18          cardlayout = new CardLayout();
19          cardPanel = new JPanel(cardlayout);
20          LevelRunner lr = new LevelRunner();
21          lr.start();
22          cardPanel.add(lr, "lr");
23      }
24  
25      @Override
26      public void run() {
27          JFrame frame = new JFrame("PacMan");
28          frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
29          frame.setVisible(true);
30          frame.getContentPane().setPreferredSize(new Dimension(960, 640));
31          frame.setResizable(false);
32          frame.add(cardPanel);
33          frame.pack();
34      }
35  
36  }
