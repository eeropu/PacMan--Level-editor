1   package pacman.gameobjects;
2   
3   import java.awt.Graphics;
4   import java.awt.Graphics2D;
5   import java.awt.Rectangle;
6   import java.awt.geom.AffineTransform;
7   import java.awt.image.BufferedImage;
8   import pacman.levelmanagement.LevelRunner;
9   
10  /**
11   * This class contains the functions of the Powerpellets. Powerpellets can be
12   * collected by PacMan. After one is collected PacMan will be able to eat the
13   * ghosts for a short period of time.
14   *
15   * @author eerop
16   */
17  public class PowerPellet implements GameObject {
18  
19      private int x, y;
20      private BufferedImage image;
21      private LevelRunner lr;
22      private AffineTransform aT;
23      private double rotation;
24  
25      /**
26       * Costructor for the PowerPellet-class.
27       *
28       * @param x x coordinate
29       * @param y y coordinate
30       */
31      public PowerPellet(int x, int y) {
32          this.x = x * 32 - 32;
33          this.y = y * 32 - 32;
34          this.aT = new AffineTransform();
35          this.rotation = 0;
36      }
37  
38      @Override
39      public void paint(Graphics g) {
40          Graphics2D g2d = (Graphics2D) g;
41          aT.rotate(0, 0, 0);
42          g2d.rotate(rotation, x + 16, y + 16);
43          g2d.drawImage(image, x, y, lr);
44          g2d.setTransform(aT);
45      }
46  
47      @Override
48      public void move() {
49          rotation += 0.1;
50          if (rotation >= Math.PI * 2) {
51              rotation = 0;
52          }
53      }
54  
55      @Override
56      public Rectangle getBounds() {
57          return new Rectangle(x + 6, y + 6, 20, 20);
58      }
59  
60      @Override
61      public boolean checkCollision(PacMan pacman) {
62          return this.getBounds().intersects(pacman.getBounds());
63      }
64  
65      public void setImage(BufferedImage img) {
66          this.image = img;
67      }
68  
69      public void setImageObserver(LevelRunner lr) {
70          this.lr = lr;
71      }
72  
73      /*
74       * Following classes are for test purposes
75       */
76      public int getX() {
77          return x;
78      }
79  
80      public int getY() {
81          return y;
82      }
83  
84      public BufferedImage getImage() {
85          return image;
86      }
87  
88      public double getRotation() {
89          return rotation;
90      }
91  
92      public void setRotation(double rotation) {
93          this.rotation = rotation;
94      }
95  
96  }
