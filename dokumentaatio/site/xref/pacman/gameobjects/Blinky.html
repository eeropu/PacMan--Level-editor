1   package pacman.gameobjects;

3   import java.awt.Color;
4   import java.awt.Graphics;
5   
6   /**
7    *
8    * @author eerop
9    *
10   * This class is responsible for managing the Blinky-ghosts. Blinky is the ghost
11   * that follows and tries to catch pacman when it gets close enough. until then
12   * it moves randomly.
13   */
14  public class Blinky extends Ghost {
15  
16      public Blinky(int x, int y, PacMan pacman, boolean randomghost) {
17          super(x, y, pacman, randomghost);
18          this.randomghost = randomghost;
19      }
20  
21      @Override
22      public void move() {
23          if (ppEaten) {
24              eatableTimer(System.currentTimeMillis());
25          }
26          if (move == 1 && !ppEaten) {
27              if (x % 32 == 0 && y % 32 == 0) {
28                  move = 2;
29              }
30          }
31          if (Math.sqrt(Math.pow(pacman.getX() - x, 2) + Math.pow(pacman.getY() - y, 2)) < 192) {
32              dSetRandomly = false;
33              if (x % 32 == 0 && y % 32 == 0) {
34                  setDirectionAStar(pacman.getX(), pacman.getY(), ppEaten);
35              }
36          } else {
37              dSetRandomly = true;
38              if (x % 32 == 0 && y % 32 == 0) {
39                  randir.randomDirection(this);
40              }
41          }
42          super.move();
43      }
44  
45      @Override
46      public void paint(Graphics g) {
47          if (ppEaten) {
48              g.setColor(Color.blue);
49              g.fillRect(x, y, 32, 32);
50          } else {
51              if (randomghost) {
52                  g.setColor(Color.green);
53                  g.fillRect(x, y, 32, 32);
54              } else {
55                  g.setColor(Color.red);
56                  g.fillRect(x, y, 32, 32);
57              }
58          }
59      }
60  }
