3   import java.awt.event.ActionEvent;
4   import javax.swing.AbstractAction;
5   import javax.swing.Action;
6   import static javax.swing.JComponent.WHEN_IN_FOCUSED_WINDOW;
7   import javax.swing.KeyStroke;
8   
9   /**
10   *
11   * @author eerop
12   *
13   * This class is used by Levelrunner and it enables controlling of PacMan.
14   */
15  public final class ControlSetUp {
16  
17      private final LevelRunner lr;
18  
19      public ControlSetUp(LevelRunner lr) {
20          this.lr = lr;
21          right();
22          left();
23          down();
24          up();
25      }
26  
27      /**
28       * Sets the action for the right arrow-key.
29       */
30      public void right() {
31          Action right = new AbstractAction() {
32  
33              @Override
34              public void actionPerformed(ActionEvent e) {
35                  lr.pacman.right();
36              }
37          };
38  
39          lr.getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("RIGHT"), "right");
40          lr.getActionMap().put("right", right);
41      }
42
