15  /**
16   *
17   * @author eerop
18   *
19   * This class is responsible for running the game. It paints all the components,
20   * keeps track of collected points, calls necessary methods from the
21   * gameobjects, check if PacMan has collided with some of the other ojects and
22   * finally sends the program to correct screen depending on player completing
23   * the level or not.
24   */
25  public class LevelRunner extends JPanel {
26  
27      private GameLoop gl;
28      protected PacMan pacman;
29      private HashSet<Wall> walls;
30      public HashSet<Ghost> ghosts;
31      private HashSet<Pointbubble> points;
32      private HashSet<PowerPellet> pp;
33      private HashMap<Integer, HashSet<Integer>> ghostStartingPositions;
34      private Timer timer;
35      private int score, lives;
36      private long now, respawn;
37      private WindowHandler wh;
38  
39      public LevelRunner(WindowHandler wh, String level) {
40          this.wh = wh;
41          build(level);
42          ControlSetUp csu = new ControlSetUp(this);
43          gl = new GameLoop(this);
44          timer = new Timer(15, gl);
45          lives = 3;
46      }
47  
48      @Override
49      public void paintComponent(Graphics g) {
50          super.paintComponent(g);
51          pacman.paint(g);
52          for (Wall wall : walls) {
53              wall.paint(g);
54          }
55          for (Ghost ghost : ghosts) {
56              ghost.paint(g);
57          }
58          for (Pointbubble point : points) {
59              point.paint(g);
60          }
61          for (PowerPellet pp1 : pp) {
62              pp1.paint(g);
63          }
64          g.setColor(Color.black);
65          g.fillRect(0, 640, 960, 32);
66          g.setColor(Color.white);
67          g.setFont(new Font("Verdana", Font.BOLD, 25));
68          g.drawString("Score: " + score, 32, 665);
69          g.fillOval(850, 640, 32, 32);
70          g.drawString("x " + lives, 900, 665);
71          if (now - respawn < 3000) {
72              g.setColor(Color.black);
73              g.fillRect(85, 275, 795, 120);
74              g.setFont(new Font("Verdana", Font.BOLD, 128));
75              g.setColor(new Color(180, 0, 0));
76              g.drawString("LIFE LOST!", 90, 384);
77          }
78      }
79  
80      public void move() {
81          now = System.currentTimeMillis();
82          if (now - respawn > 3000) {
83              pacman.move();
84              for (Ghost ghost : ghosts) {
85                  ghost.move();
86              }
87          }
88      }
89  
90      public void checkCollision() {
91          for (Wall wall : walls) {
92              if (wall.checkCollision(pacman)) {
93                  pacman.stop();
94              }
95          }
96          Iterator<Pointbubble> pbit = points.iterator();
97          while (pbit.hasNext()) {
98              if (pbit.next().checkCollision(pacman)) {
99                  pbit.remove();
100                 score += 10;
101             }
102         }
103         if (points.isEmpty()) {
104             completed();
105         }
106         Iterator<PowerPellet> ppit = pp.iterator();
107         while (ppit.hasNext()) {
108             if (ppit.next().checkCollision(pacman)) {
109                 ppit.remove();
110                 score += 50;
111                 for (Ghost ghost : ghosts) {
112                     ghost.eatPowerpellet(System.currentTimeMillis());
113                 }
114             }
115         }
116         for (Ghost ghost : ghosts) {
117             for (Wall wall : walls) {
118                 if (ghost.getBounds().intersects(wall.getBounds())) {
119                     ghost.stop();
120                 }
121             }
122             if (ghost.checkCollision(pacman)) {
123                 if (ghost.isPpEaten()) {
124                     PriorityQueue<Coordinate> pq = new PriorityQueue();
125                     for (Integer i : ghostStartingPositions.keySet()) {
126                         for (Integer j : ghostStartingPositions.get(i)) {
127                             pq.add(new Coordinate(i, j, 0, pacman.getX(), pacman.getY(), null, true));
128                         }
129                     }
130                     Coordinate farthest = pq.poll();
131                     ghost.setX(farthest.getX());
132                     ghost.setY(farthest.getY());
133                     ghost.setPpEaten(false);
134                     score += 200;
135                 } else {
136                     pacman.reset();
137                     for (Ghost g : ghosts) {
138                         g.reset();
139                     }
140                     respawn = System.currentTimeMillis();
141                     lives--;
142                 }
143             }
144         }
145     }
146 
147     public void start() {
148         timer.start();
149     }
150 
151     public void build(String s) {
152         LevelBuilder lb = new LevelBuilder(s);
153         lb.build();
154         pacman = lb.getPacman();
155         walls = lb.getWalls();
156         ghosts = lb.getGhosts();
157         points = lb.getPoints();
158         pp = lb.getPp();
159         ghostStartingPositions = new HashMap<>();
160         for (Ghost ghost : ghosts) {
161             if (ghostStartingPositions.containsKey(ghost.getX())) {
162                 ghostStartingPositions.get(ghost.getX()).add(ghost.getY());
163             } else {
164                 ghostStartingPositions.put(ghost.getX(), new HashSet<>());
165                 ghostStartingPositions.get(ghost.getX()).add(ghost.getY());
166             }
167         }
168     }
169 
170     public void completed() {
171         timer.stop();
172         wh.lvlCompleted();
173     }
