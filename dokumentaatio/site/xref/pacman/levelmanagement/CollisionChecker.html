3   import java.util.HashMap;
4   import java.util.HashSet;
5   import java.util.Iterator;
6   import java.util.PriorityQueue;
7   import pacman.gameobjects.Coordinate;
8   import pacman.gameobjects.Ghost;
9   import pacman.gameobjects.PacMan;
10  import pacman.gameobjects.Pointbubble;
11  import pacman.gameobjects.PowerPellet;
12  import pacman.gameobjects.Wall;
13  
14  /**
15   * Used to check if components collide with pacman
16   *
17   * @author eerop
18   */
19  public class CollisionChecker {
20  
21      private LevelRunner lr;
22      private PacMan pacman;
23      private HashSet<Wall> walls;
24      private HashSet<Ghost> ghosts;
25      private HashSet<Pointbubble> points;
26      private HashSet<PowerPellet> pp;
27      private HashMap<Integer, HashSet<Integer>> ghostStartingPositions;
28      private GameLoop gl;
29  
30      public CollisionChecker(LevelRunner lr, GameLoop gl, PacMan pacman, HashSet<Wall> walls, HashSet<Ghost> ghosts,
31              HashSet<Pointbubble> points, HashSet<PowerPellet> pp, HashMap<Integer, HashSet<Integer>> ghostStartingPositions) {
32          this.lr = lr;
33          this.gl = gl;
34          this.pacman = pacman;
35          this.walls = walls;
36          this.ghosts = ghosts;
37          this.points = points;
38          this.pp = pp;
39          this.ghostStartingPositions = ghostStartingPositions;
40      }
41  
42      public void checkCollision() {
43          for (Wall wall : walls) {
44              if (wall.checkCollision(pacman)) {
45                  pacman.stop();
46              }
47          }
48          Iterator<Pointbubble> pbit = points.iterator();
49          while (pbit.hasNext()) {
50              if (pbit.next().checkCollision(pacman)) {
51                  pbit.remove();
52                  lr.addScore(10);
53              }
54          }
55          if (points.isEmpty()) {
56              lr.completed();
57          }
58          Iterator<PowerPellet> ppit = pp.iterator();
59          while (ppit.hasNext()) {
60              if (ppit.next().checkCollision(pacman)) {
61                  ppit.remove();
62                  lr.addScore(50);
63                  for (Ghost ghost : ghosts) {
64                      ghost.eatPowerpellet(System.currentTimeMillis());
65                  }
66              }
67          }
68          for (Ghost ghost : ghosts) {
69              for (Wall wall : walls) {
70                  if (ghost.getBounds().intersects(wall.getBounds())) {
71                      ghost.stop();
72                  }
73              }
74              if (ghost.checkCollision(pacman)) {
75                  if (ghost.isPpEaten()) {
76                      PriorityQueue<Coordinate> pq = new PriorityQueue();
77                      for (Integer i : ghostStartingPositions.keySet()) {
78                          for (Integer j : ghostStartingPositions.get(i)) {
79                              pq.add(new Coordinate(i, j, 0, pacman.getX(), pacman.getY(), null, true));
80                          }
81                      }
82                      Coordinate farthest = pq.poll();
83                      ghost.setX(farthest.getX());
84                      ghost.setY(farthest.getY());
85                      ghost.setPpEaten(false);
86                      lr.addScore(200);
87                  } else {
88                      pacman.reset();
89                      for (Ghost g : ghosts) {
90                          g.reset();
91                      }
92                      gl.setRespawn(System.currentTimeMillis());
93                      lr.loseLife();
94                      lr.setRespawning(true);
95                  }
96              }
97          }
98      }
99  
100 }
