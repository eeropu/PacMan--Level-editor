1   package pacman.guiobjects;
2   
3   import java.awt.Color;
4   import java.awt.Font;
5   import java.awt.Graphics;
6   import java.awt.image.BufferedImage;
7   import javax.swing.JButton;
8   import javax.swing.JPanel;
9   import pacman.guilisteners.CreationMenuListener;
10  import pacman.pacman.leveleditor.ImageGetter;
11  
12  /**
13   * This class manages the creationmenu, that allows user to create a new level
14   * or modify an old one.
15   *
16   * @author eerop
17   */
18  public class CreationMenu extends JPanel {
19  
20      private BufferedImage image;
21      private JButton createNew, modifyOld, back, delete;
22      private CreationMenuListener cml;
23      private ImageGetter imgGetter;
24  
25      /**
26       * Constructor for the CreationMenu-class.
27       *
28       * @param cml CreationMenuListener that gives functionality for the buttons
29       * in this menu
30       * @param imgGetter used to get all the graphical components in this screen
31       */
32      public CreationMenu(CreationMenuListener cml, ImageGetter imgGetter) {
33          this.cml = cml;
34          this.imgGetter = imgGetter;
35          this.image = imgGetter.getImage("Pictures/PacMan_Createmode.png");
36          repaint();
37  
38          setLayout(null);
39  
40          Color c = new Color(87, 0, 127);
41          createNew = new JButton("Create New!");
42          createNew.setBounds(64, 64, 320, 128);
43          createNew.setFont(new Font("Verdana", Font.BOLD, 30));
44          createNew.setBackground(c);
45          createNew.setForeground(Color.black);
46  
47          modifyOld = new JButton("Modify Existing!");
48          modifyOld.setBounds(64, 192, 320, 128);
49          modifyOld.setFont(new Font("Verdana", Font.BOLD, 30));
50          modifyOld.setBackground(c);
51          modifyOld.setForeground(Color.black);
52  
53          delete = new JButton("Delete");
54          delete.setBackground(c);
55          delete.setForeground(Color.black);
56  
57          delete.setBounds(96, 352, 256, 32);
58  
59          back = new JButton("back");
60          back.setBounds(64, 608, 96, 32);
61          back.setBackground(c);
62          back.setForeground(Color.black);
63  
64          cml.setButtons(createNew, modifyOld, back, delete);
65          modifyOld.addActionListener(cml);
66          back.addActionListener(cml);
67          createNew.addActionListener(cml);
68          delete.addActionListener(cml);
69  
70          add(createNew);
71          add(modifyOld);
72          add(back);
73          add(delete);
74      }
75  
76      @Override
77      public void paintComponent(Graphics g) {
78          g.drawImage(image, 0, 0, this);
79      }
80  }
