6   package pacman.guiobjects;
7   
8   import java.awt.Color;
9   import java.awt.Font;
10  import java.awt.Graphics;
11  import java.awt.GridLayout;
12  import java.awt.image.BufferedImage;
13  import java.io.IOException;
14  import java.net.URL;
15  import javax.imageio.ImageIO;
16  import javax.swing.JButton;
17  import javax.swing.JLabel;
18  import javax.swing.JPanel;
19  import javax.swing.JTextPane;
20  import javax.swing.text.SimpleAttributeSet;
21  import javax.swing.text.StyleConstants;
22  import javax.swing.text.StyledDocument;
23  import pacman.guilisteners.LevelCompleteListener;
24  
25  /**
26   *
27   * @author eerop
28   *
29   * Menu that will be shown to the player once a level is finished.
30   */
31  public class LevelCompleted extends JPanel {
32  
33      private BufferedImage image;
34      private JLabel label;
35      private JTextPane txt;
36      private JButton submit, skip;
37      private LevelCompleteListener lcl;
38  
39      public LevelCompleted(LevelCompleteListener lcl) {
40  
41          this.lcl = lcl;
42  
43          ClassLoader cl = this.getClass().getClassLoader();
44          URL resource = cl.getResource("Pictures/PacMan_lvlcomplete.png");
45          try {
46              image = ImageIO.read(resource);
47          } catch (IOException e) {
48          }
49          repaint();
50          setLayout(null);
51  
52          label = new JLabel("Submit your score!");
53          label.setFont(new Font("Verdana", Font.BOLD, 25));
54          label.setForeground(Color.white);
55          label.setHorizontalAlignment((int) CENTER_ALIGNMENT);
56          label.setBounds(320, 320, 320, 40);
57          add(label);
58  
59          JPanel panel = new JPanel(new GridLayout(2, 1));
60          panel.setBounds(320, 360, 320, 80);
61          panel.setOpaque(true);
62  
63          txt = new JTextPane();
64          txt.setFont(new Font("Verdana", Font.BOLD, 25));
65          txt.setBackground(new Color(20, 20, 20));
66          txt.setForeground(Color.gray);
67          txt.setText("Name");
68          StyledDocument doc = txt.getStyledDocument();
69          SimpleAttributeSet center = new SimpleAttributeSet();
70          StyleConstants.setAlignment(center, StyleConstants.ALIGN_CENTER);
71          doc.setParagraphAttributes(0, doc.getLength(), center, false);
72          lcl.setTxt(txt);
73          txt.addMouseListener(lcl);
74          panel.add(txt);
75  
76          submit = new JButton("Submit!");
77          submit.setFont(new Font("Verdana", Font.BOLD, 20));
78          submit.setForeground(Color.black);
79          lcl.setSubmit(submit);
80          submit.addActionListener(lcl);
81          panel.add(submit);
82  
83          skip = new JButton("Skip");
84          skip.setBounds(450, 608, 64, 32);
85          lcl.setSkip(skip);
86          skip.addActionListener(lcl);
87  
88          add(panel);
89          add(skip);
90      }
91  
92      @Override
93      public void paintComponent(Graphics g) {
94          g.drawImage(image, 0, 0, this);
95      }
96  }
