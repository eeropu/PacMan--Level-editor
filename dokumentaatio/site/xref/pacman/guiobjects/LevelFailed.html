1   package pacman.guiobjects;
2   
3   import java.awt.Font;
4   import java.awt.Graphics;
5   import java.awt.event.ActionEvent;
6   import java.awt.event.ActionListener;
7   import java.awt.image.BufferedImage;
8   import javax.swing.JButton;
9   import javax.swing.JPanel;
10  import pacman.pacman.leveleditor.ImageGetter;
11  import pacman.pacman.leveleditor.WindowHandler;
12  
13  /**
14   * Screen that is shown to the player if he/she loses the level.
15   *
16   * @author eerop
17   */
18  public class LevelFailed extends JPanel {
19  
20      private JButton button;
21      private ImageGetter imgGetter;
22      private BufferedImage image;
23      private WindowHandler wh;
24  
25      public LevelFailed(ImageGetter imgGetter, WindowHandler wh) {
26          this.wh = wh;
27          this.imgGetter = imgGetter;
28          this.image = imgGetter.getImage("Pictures/PacMan_lvlfailed.png");
29          repaint();
30  
31          setLayout(null);
32  
33          this.button = new JButton("Back to Menu");
34          button.setFont(new Font("Verdana", Font.BOLD, 20));
35          button.setBounds(704, 576, 224, 64);
36          button.addActionListener(new ActionListener() {
37  
38              @Override
39              public void actionPerformed(ActionEvent e) {
40                  wh.startMenu();
41              }
42          });
43  
44          add(button);
45      }
46  
47      @Override
48      public void paintComponent(Graphics g) {
49          g.drawImage(image, 0, 0, this);
50      }
51  
52  }
