1   package pacman.guiobjects;
2   
3   import java.awt.Font;
4   import javax.swing.JButton;
5   import javax.swing.JLabel;
6   import javax.swing.JPanel;
7   import javax.swing.JRadioButton;
8   import javax.swing.JTextPane;
9   import pacman.guilisteners.FinishCreationListener;
10  import pacman.pacman.leveleditor.WindowHandler;
11  
12  /**
13   * Menu where user can define the name and timeframe of a level and fill it with
14   * pointbubbles.
15   *
16   * @author eerop
17   */
18  public class FinishCreation extends JPanel {
19  
20      private String objectPositioning;
21      private JTextPane txtname, txtlives, txttime, pPerSec;
22      private Font font;
23      private JRadioButton xtrapoints, deadline, fillbubbles;
24      private JButton button;
25      private WindowHandler wh;
26  
27      /**
28       * Constructor for the FinishCreation-class.
29       *
30       * @param objectPositioning string representation of the objectpositioning
31       * for the level that is being created or modified
32       * @param wh windowhandler that is used to change the contents on the screen
33       */
34      public FinishCreation(String objectPositioning, WindowHandler wh) {
35          this.objectPositioning = objectPositioning;
36          this.wh = wh;
37          setLayout(null);
38          createLabels();
39          txtname = new JTextPane();
40          txtname.setBounds(608, 35, 320, 26);
41          txtname.setFont(new Font("Verdana", Font.BOLD, 15));
42          txtlives = new JTextPane();
43          txtlives.setBounds(736, 128, 64, 32);
44          txtlives.setFont(font);
45          txttime = new JTextPane();
46          txttime.setBounds(736, 240, 64, 32);
47          txttime.setFont(font);
48          pPerSec = new JTextPane();
49          pPerSec.setBounds(384, 352, 64, 32);
50          pPerSec.setFont(font);
51          add(txtname);
52          add(txtlives);
53          add(txttime);
54          add(pPerSec);
55  
56          xtrapoints = new JRadioButton();
57          xtrapoints.setBounds(288, 304, 32, 32);
58          add(xtrapoints);
59          deadline = new JRadioButton();
60          deadline.setBounds(655, 304, 32, 32);
61          add(deadline);
62          fillbubbles = new JRadioButton();
63          fillbubbles.setBounds(544, 448, 32, 32);
64          add(fillbubbles);
65  
66          button = new JButton("FINISH!");
67          button.setFont(new Font("Verdana", Font.BOLD, 35));
68          button.setBounds(320, 544, 320, 96);
69          add(button);
70  
71          FinishCreationListener fcl = new FinishCreationListener(txtname, txtlives, txttime, pPerSec,
72                  xtrapoints, deadline, fillbubbles, button, wh, objectPositioning);
73          button.addActionListener(fcl);
74      }
75  
76      private void createLabels() {
77          font = new Font("Verdana", Font.BOLD, 20);
78          JLabel name = new JLabel("Give your level a name! (MAX 30 characters)");
79          name.setFont(font);
80          name.setBounds(32, 32, 512, 32);
81          add(name);
82          JLabel lives = new JLabel("Enter the numer of lives when starting this level (0 - 9):");
83          lives.setFont(font);
84          lives.setBounds(32, 128, 704, 32);
85          add(lives);
86          JLabel time = new JLabel("<html><p>Enter the time frame (seconds: 0 - 300) <br> in which this level needs to be completed to:</p><html>");
87          time.setFont(font);
88          time.setBounds(32, 224, 704, 64);
89          add(time);
90          JLabel time2 = new JLabel("Gain extra points: ");
91          time2.setFont(font);
92          time2.setBounds(64, 304, 224, 32);
93          add(time2);
94          JLabel time3 = new JLabel("Complete the level: ");
95          time3.setFont(font);
96          time3.setBounds(416, 304, 224, 32);
97          add(time3);
98          JLabel pointPerSec = new JLabel("Points / second (1 - 100):");
99          pointPerSec.setFont(font);
100         pointPerSec.setBounds(64, 352, 320, 32);
101         add(pointPerSec);
102         JLabel fill = new JLabel("Fill empty coordinates with pointbubbles:");
103         fill.setFont(font);
104         fill.setBounds(32, 448, 544, 32);
105         add(fill);
106     }
